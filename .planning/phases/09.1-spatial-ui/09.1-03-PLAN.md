Create a reusable Bento card component with Liquid Glass mouse-tracking effect.

This is a **client component** because it needs mouse event handling.

```tsx
'use client'

import { useRef, type ReactNode, type MouseEvent } from 'react'

interface BentoCardProps {
  children: ReactNode
  className?: string
  size?: 'large' | 'small'
}

export function BentoCard({ children, className = '', size = 'large' }: BentoCardProps) {
  const cardRef = useRef<HTMLDivElement>(null)

  const handleMouseMove = (e: MouseEvent<HTMLDivElement>) => {
    if (!cardRef.current) return
    const rect = cardRef.current.getBoundingClientRect()
    const x = ((e.clientX - rect.left) / rect.width) * 100
    const y = ((e.clientY - rect.top) / rect.height) * 100
    cardRef.current.style.setProperty('--mouse-x', `${x}%`)
    cardRef.current.style.setProperty('--mouse-y', `${y}%`)
  }

  const handleMouseLeave = () => {
    if (!cardRef.current) return
    cardRef.current.style.setProperty('--mouse-x', '50%')
    cardRef.current.style.setProperty('--mouse-y', '50%')
  }

  return (
    <div
      ref={cardRef}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      className={`
        relative overflow-hidden rounded-2xl
        bg-[var(--surface)] border border-[var(--platinum)]/10
        angle-glow-ambient
        ${size === 'large' ? 'p-8' : 'p-6'}
        ${className}
      `}
      style={{
        '--mouse-x': '50%',
        '--mouse-y': '50%',
      } as React.CSSProperties}
    >
      {/* Liquid Glass overlay */}
      <div
        className="absolute inset-0 pointer-events-none opacity-0 hover-parent:opacity-100 transition-opacity duration-300"
        style={{
          background: `radial-gradient(
            circle at var(--mouse-x) var(--mouse-y),
            rgba(255, 255, 255, 0.06) 0%,
            transparent 50%
          )`,
        }}
      />

      {/* Content */}
      <div className="relative z-10">{children}</div>
    </div>
  )
}
```

Also add the liquid-glass CSS to globals.css:
```css
/* Liquid Glass Effect */
.bento-card {
  --mouse-x: 50%;
  --mouse-y: 50%;
}

.bento-card::after {
  content: '';
  position: absolute;
  inset: 0;
  background: radial-gradient(
    circle at var(--mouse-x) var(--mouse-y),
    rgba(255, 255, 255, 0.06) 0%,
    transparent 50%
  );
  pointer-events: none;
  opacity: 0;
  transition: opacity 0.3s ease;
}

.bento-card:hover::after {
  opacity: 1;
}
```
